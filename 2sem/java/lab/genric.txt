import java.util.ArrayList;
import java.util.List;

class NumberSearch<T extends Number> {
    private List<T> numbers;

    public NumberSearch(List<T> numbers) {
        this.numbers = numbers;
    }

    public boolean search(T target) {
        for (T num : numbers) {
            if (num.equals(target)) {
                return true;
            }
        }
        return false;
    }
}

public class NumberSearchExample {
    public static void main(String[] args) {
        List<Integer> integerList = new ArrayList<>();
        integerList.add(1);
        integerList.add(2);
        integerList.add(3);

        NumberSearch<Integer> integerSearch = new NumberSearch<>(integerList);

        int targetInteger = 2;
        boolean foundInteger = integerSearch.search(targetInteger);
        System.out.println("Found " + targetInteger + ": " + foundInteger);

        List<Double> doubleList = new ArrayList<>();
        doubleList.add(1.5);
        doubleList.add(2.5);
        doubleList.add(3.5);

        NumberSearch<Double> doubleSearch = new NumberSearch<>(doubleList);

        double targetDouble = 2.5;
        boolean foundDouble = doubleSearch.search(targetDouble);
        System.out.println("Found " + targetDouble + ": " + foundDouble);
    }
}
