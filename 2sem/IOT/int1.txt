int LED1=2;
int MQ=A0;
int LDR=A1;

void setup()
{
  pinMode(LED1,OUTPUT);
  pinMode(MQ,INPUT);
  pinMode(LDR,INPUT);
  Serial.begin(9600);
}
void loop()
{
  int mqval=analogRead(MQ);
  int ldrval=analogRead(LDR);
  Serial.print(mqval);
  Serial.print(" ");
  Serial.println(ldrval);
  if(mqval > 100 && ldrval < 400)
  {
    digitalWrite(LED1,HIGH);
    Serial.println("Gag Leakage");
  }
  else 
  {
    digitalWrite(LED1,LOW);
    Serial.println("No gas Leakage");
  }
  delay(500);
}
